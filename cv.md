```
function duplicateEncode(word){
  const wordArray = word.toLowerCase().split('');
  let code = '';
  for (let i = 0; i < wordArray.length; i++) {
    const filtered = wordArray.filter((char) => char === word.toLowerCase()[i]);
    filtered.length > 1 ? code += ')' : code += '(';
  }
  return code;
}
```
